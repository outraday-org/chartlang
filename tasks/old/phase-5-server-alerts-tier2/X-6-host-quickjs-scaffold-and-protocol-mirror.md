### 2. `packages/host-quickjs/src/protocol.ts` (new)

Two-line MIT header, then mirror host-worker's protocol:

```ts
import type {
    AdapterSymInfo,
    CandleEvent,
    Capabilities,
    RunnerEmissions,
} from "@invinite-org/chartlang-adapter-kit";

import type { QuickJsCompiledScript, QuickJsHostLimits } from "./types";

/**
 * Messages the host posts into the QuickJS runtime. JSON-clean by
 * construction — every payload survives the QuickJS membrane (Task 7)
 * without bespoke transferables.
 *
 * Mirrors `@invinite-org/chartlang-host-worker`'s `HostToWorker`
 * byte-for-byte (same field names + types). The two protocols are
 * deliberately interchangeable so cross-host determinism tests can
 * dispatch the same frame to either host.
 *
 * @since 0.5
 * @example
 *     const m: HostToQuickJs = { kind: "dispose" };
 *     void m;
 */
export type HostToQuickJs =
    | {
          readonly kind: "load";
          readonly compiled: QuickJsCompiledScript;
          readonly capabilities: Capabilities;
          readonly symInfo?: AdapterSymInfo;
          readonly inputOverrides?: Readonly<Record<string, unknown>>;
          readonly limits: QuickJsHostLimits;
      }
    | { readonly kind: "candleEvent"; readonly event: CandleEvent }
    | { readonly kind: "drain"; readonly nonce: number }
    | { readonly kind: "dispose" };

export type QuickJsToHost =
    | { readonly kind: "loaded" }
    | { readonly kind: "loadError"; readonly message: string }
    | {
          readonly kind: "emissions";
          readonly nonce: number;
          readonly emissions: RunnerEmissions;
      }
    | { readonly kind: "step-overshoot"; readonly observedMs: number }
    | { readonly kind: "fatal"; readonly message: string };
```

### 3. `packages/host-quickjs/src/types.ts` (new)

Two-line MIT header, then:

```ts
import type {
    AdapterSymInfo,
    Capabilities,
    RunnerEmissions,
} from "@invinite-org/chartlang-adapter-kit";

/**
 * Subset of `CompiledScript` (per `@invinite-org/chartlang-compiler`)
 * the QuickJS host needs. Matches host-worker's `HostCompiledScript`
 * shape verbatim.
 *
 * @since 0.5
 */
export type QuickJsCompiledScript = Readonly<{
    moduleSource: string;
    manifest: import("@invinite-org/chartlang-core").ScriptManifest;
}>;

/**
 * Hard runtime caps. `maxHeapBytes` defaults to 64 MB per PLAN §8.3;
 * `maxStepMs` defaults to 1 ms (the CPU budget per `compute` step).
 *
 * @since 0.5
 */
export type QuickJsHostLimits = Readonly<{
    maxHeapBytes: number;
    maxStepMs: number;
}>;

/**
 * Injection seam for tests. Production callers omit; the host then
 * boots a real `quickjs-emscripten` context.
 *
 * @since 0.5
 */
export type QuickJsLike = {
    eval(source: string): Promise<unknown>;
    dispose(): void;
};

/**
 * Mirrors `host-worker`'s `ScriptHost` shape for cross-host
 * interchangeability. Re-exports the runtime's type alias to keep
 * the symbol identity stable across packages.
 *
 * @since 0.5
 */
export type ScriptHost = import("@invinite-org/chartlang-host-worker").ScriptHost;
```

(If `host-worker` doesn't yet re-export `ScriptHost` from its
public barrel, add the re-export there as part of this task — the
type lives in `host-worker/src/types.ts`; lifting it to the public
surface is additive and stable.)
